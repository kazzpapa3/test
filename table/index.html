	.list-table {
		border: 1px solid #ccc;

	}
	.list-table td {
		display: block;
		position: relative;
		padding: .4em .4em .4em 7em; //このpaddingのleftの数値の設定で左側に来るdata-thの横幅分ずらしてくれる
